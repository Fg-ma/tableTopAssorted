***install
sudo apt update && sudo apt upgrade -y
sudo apt install -y curl wget lsb-release gnupg
sudo apt install -y podman
curl --silent --remote-name --location https://download.ceph.com/rpm-19.2.0/el9/noarch/cephadm
chmod +x cephadm
sudo mv cephadm /usr/local/bin/
cephadm version
sudo cephadm add-repo --release squid
sudo cephadm install
sudo apt install -y ceph cephadm ceph-common ceph-fuse ceph-mgr ceph-mon ceph-osd
sudo cephadm bootstrap --mon-ip 127.0.0.1 --skip-mon-network 

---------if needed-----------
sudo mkdir -p /etc/containers
sudo tee /etc/containers/policy.json > /dev/null <<EOF
{
  "default": [
    {
      "type": "insecureAcceptAnything"
    }
  ]
}
EOF
